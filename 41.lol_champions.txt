CREATE TABLE public.lol_champions
(
    id integer NOT NULL DEFAULT nextval('lol_champions_id_seq'::regclass),
    champion_id integer,
    armor integer,
    armorperlevel integer,
    attackdamage integer,
    attackdamageperlevel integer,
    attackrange integer,
    attackspeedoffset integer,
    attackspeedperlevel double precision,
    big_image_url character varying(255) COLLATE pg_catalog."default",
    crit integer,
    critperlevel integer,
    hp integer,
    hpperlevel integer,
    hpregen integer,
    hpregenperlevel double precision,
    image_url character varying(255) COLLATE pg_catalog."default",
    movespeed integer,
    mp integer,
    mpperlevel integer,
    mpregen integer,
    mpregenperlevel integer,
    name character varying(255) COLLATE pg_catalog."default",
    spellblock integer,
    spellblockperlevel double precision,
    created_at timestamp with time zone NOT NULL,
    updated_at timestamp with time zone NOT NULL,
    CONSTRAINT lol_champions_pkey PRIMARY KEY (id)
)
