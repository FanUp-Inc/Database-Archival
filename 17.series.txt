-- Table: public.series

-- DROP TABLE public.series;

CREATE TABLE public.series
(
    id integer NOT NULL DEFAULT nextval('series_id_seq'::regclass),
    serie_id integer,
    league_id integer,
    winner_id integer,
    winner_type text COLLATE pg_catalog."default",
    name text COLLATE pg_catalog."default",
    full_name text COLLATE pg_catalog."default",
    description text COLLATE pg_catalog."default",
    season text COLLATE pg_catalog."default",
    slug text COLLATE pg_catalog."default",
    year integer,
    begin_at timestamp with time zone,
    end_at timestamp with time zone,
    modified_at timestamp with time zone,
    created_at timestamp with time zone NOT NULL,
    updated_at timestamp with time zone NOT NULL,
    CONSTRAINT series_pkey PRIMARY KEY (id),
    CONSTRAINT series_serie_id_key UNIQUE (serie_id),
    CONSTRAINT series_league_id_fkey FOREIGN KEY (league_id)
        REFERENCES public."Leagues" (league_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE public.series
    OWNER to backend;
