-- DROP TABLE public.matches_teams;

CREATE TABLE public.matches_teams
(
    id integer NOT NULL DEFAULT nextval('matches_teams_id_seq'::regclass),
    match_id integer,
    team_id integer,
    created_at timestamp with time zone NOT NULL,
    updated_at timestamp with time zone NOT NULL,
    CONSTRAINT matches_teams_pkey PRIMARY KEY (id),
    CONSTRAINT matches_teams_match_id_fkey FOREIGN KEY (match_id)
        REFERENCES public."Matches" (ps_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT matches_teams_team_id_fkey FOREIGN KEY (team_id)
        REFERENCES public."Teams" (ps_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE public.matches_teams
    OWNER to backend;
