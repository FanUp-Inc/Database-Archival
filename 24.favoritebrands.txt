CREATE TABLE public."FavoriteBrands"
(
    id integer NOT NULL DEFAULT nextval('"FavoriteBrands_id_seq"'::regclass),
    user_id integer NOT NULL,
    brand_id integer NOT NULL,
    status integer,
    "createdAt" timestamp with time zone NOT NULL,
    "updatedAt" timestamp with time zone NOT NULL,
    CONSTRAINT "FavoriteBrands_pkey" PRIMARY KEY (id),
    CONSTRAINT "FavoriteBrands_brand_id_fkey" FOREIGN KEY (brand_id)
        REFERENCES public."Brands" (id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION,
    CONSTRAINT "FavoriteBrands_user_id_fkey" FOREIGN KEY (user_id)
        REFERENCES public."Users" (id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE public."FavoriteBrands"
    OWNER to backend;
