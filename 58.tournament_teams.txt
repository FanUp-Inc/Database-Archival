CREATE TABLE public.tournaments_teams
(
    id integer NOT NULL DEFAULT nextval('tournaments_teams_id_seq'::regclass),
    team_id integer,
    tournament_id integer,
    created_at timestamp with time zone NOT NULL,
    updated_at timestamp with time zone NOT NULL,
    CONSTRAINT tournaments_teams_pkey PRIMARY KEY (id),
    CONSTRAINT tournaments_teams_team_id_fkey FOREIGN KEY (team_id)
        REFERENCES public."Teams" (ps_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT tournaments_teams_tournament_id_fkey FOREIGN KEY (tournament_id)
        REFERENCES public.tournaments (tournament_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE public.tournaments_teams
    OWNER to backend;
