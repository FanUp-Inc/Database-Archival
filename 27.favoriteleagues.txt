CREATE TABLE public."FavoriteLeagues"
(
    id integer NOT NULL DEFAULT nextval('"FavoriteLeagues_id_seq"'::regclass),
    user_id integer NOT NULL,
    league_id integer NOT NULL,
    status integer,
    "createdAt" timestamp with time zone NOT NULL,
    "updatedAt" timestamp with time zone NOT NULL,
    CONSTRAINT "FavoriteLeagues_pkey" PRIMARY KEY (id),
    CONSTRAINT "FavoriteLeagues_league_id_fkey" FOREIGN KEY (league_id)
        REFERENCES public."Leagues" (id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION,
    CONSTRAINT "FavoriteLeagues_user_id_fkey" FOREIGN KEY (user_id)
        REFERENCES public."Users" (id) MATCH SIMPLE
        ON UPDATE CASCADE
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE public."FavoriteLeagues"
    OWNER to backend;
